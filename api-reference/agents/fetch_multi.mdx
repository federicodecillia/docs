<RequestExample>

```bash Curl
curl --location --request GET 'https://app.gptchatbot.it/api/v1/chatbot/{uuid}/agents' \
--header 'Authorization: Bearer <token>'
```

```py Python
import requests

uuid = '<chatbot-uuid>'
url = f'https://app.gptchatbot.it/api/v1/chatbot/{uuid}/agents'
headers = {
    'Authorization': 'Bearer <token>'
}

response = requests.get(url, headers=headers)

if response.status_code == 200:
    print("Request successful!")
    print(response.json())
else:
    print("Request failed with status code:", response.status_code)
    print(response.text)
```

```ts JavaScript
const axios = require("axios");

const uuid = "<chatbot-uuid>";
const url = `https://app.gptchatbot.it/api/v1/chatbot/${uuid}/agents`;
const headers = {
  Authorization: "Bearer <token>",
};

axios
  .get(url, { headers })
  .then((response) => {
    console.log("Request successful!");
    console.log(response.data);
  })
  .catch((error) => {
    console.error("Request failed:", error);
  });
```
