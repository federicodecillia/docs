
```py Python
import requests

uuid = '<chatbot-uuid>'
url = f'https://app.gptchatbot.it/api/v1/chatbot/{uuid}/source-tags'
headers = {
    'Authorization': 'Bearer <token>'
}

response = requests.get(url, headers=headers)

if response.status_code == 200:
    print("Request successful!")
    print(response.json())
else:
    print("Request failed with status code:", response.status_code)
    print(response.text)
```

```ts JavaScript
const axios = require("axios");

const url = `https://app.gptchatbot.it/api/v1/chatbot/${uuid}/source-tags`;

  });
```
